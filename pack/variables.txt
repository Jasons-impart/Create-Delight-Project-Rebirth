# 服务端启动变量，供 start.bat 和 start.sh 使用。
# Server launcher variables for start.bat and start.sh.
#
# 如果 Java 21 已在 PATH 中，保持 java 即可；也可以填写完整路径。
# Use java when Java 21 is already in PATH, or set a full path.
#
# Windows 示例 / Windows example:
# JAVA="C:\Program Files\Java\jdk-21\bin\java.exe"
#
# Linux/macOS 示例 / Linux/macOS example:
# JAVA=/usr/lib/jvm/java-21-openjdk/bin/java
JAVA=java

# 调整服务端内存，主要修改 -Xms 和 -Xmx。
# Adjust -Xms and -Xmx for the target server memory.
JVM_ARGS=-Xms4G -Xmx6G -XX:+UseG1GC -XX:+AlwaysPreTouch

# 服务端行为开关。
# Server behavior switches.
#
# 设置为 false 时，服主需要手动接受 eula.txt。
# Set ACCEPT_EULA=false if you want server operators to accept eula.txt manually.
ACCEPT_EULA=true

# 设置为 true 时，服务端退出后会自动重启。
# Set AUTO_RESTART=true to restart the server after it exits.
AUTO_RESTART=false
RESTART_DELAY_SECONDS=10

# 整合包与运行时版本。需要和 pack.toml、NeoForge installer 保持一致。
# Pack/runtime versions. Keep these aligned with pack.toml and the NeoForge installer.
RECOMMENDED_JAVA_VER=21
MC_VERSION=1.21.1
NEOFORGE_VERSION=21.1.242
NEOFORGE_INSTALLER=neoforge.jar
NEOFORGE_INSTALLER_URL=https://maven.neoforged.net/releases/net/neoforged/neoforge/21.1.242/neoforge-21.1.242-installer.jar
